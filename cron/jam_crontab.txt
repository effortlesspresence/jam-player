# JAM Player Crontab
# This file is installed by jam-update.service
# Only contains essential scheduled tasks

# Reboot the device every morning at 3am
# This ensures updates are applied and clears any accumulated issues
0 3 * * * sleep 42; /bin/systemctl reboot

# Rotate logs 3x daily to prevent disk space issues
# Runs at 1:30am, 9:30am, and 5:30pm
30 1 * * * /usr/sbin/logrotate /etc/jam/logrotate.conf 2>/dev/null || true
30 9 * * * /usr/sbin/logrotate /etc/jam/logrotate.conf 2>/dev/null || true
30 17 * * * /usr/sbin/logrotate /etc/jam/logrotate.conf 2>/dev/null || true
